53  /**
54   * Wrapper for W3C Validator Soap responses 
55   * see <a href="http://validator.w3.org/docs/api.html#requestformat">W3C Markup Validation Service Request Format</a>
56   * @author wf
57   *
58   * the JaxB annotations of this Wrapper are designed to be able to unmarshal a W3C Markup Validation Service
59   * Soap 1.2 response to this Java Object
60   * 
61   * on the class level the root node of the SOAP message is covered:
62   * <pre>
63   * {@code
64   * <env:Envelope xmlns:env="http://www.w3.org/2003/05/soap-envelope">
65   * ...
66   * </env:Envelope>
67   * }
68   * </pre>
69   *  According to <a href="http://validator.w3.org/docs/api.html#requestformat">W3C Markup Validation Service Request Format documentation</a>:
70   *  A SOAP response for the validation of a document (invalid) will look like this:
71   * <pre>
72   * {@code
73   *  <?xml version="1.0" encoding="UTF-8"?>
74   *    <env:Envelope xmlns:env="http://www.w3.org/2003/05/soap-envelope">
75   *      <env:Body>
76   *        <m:markupvalidationresponse
77   *          env:encodingStyle="http://www.w3.org/2003/05/soap-encoding" 
78   *          xmlns:m="http://www.w3.org/2005/10/markup-validator">
79   *            <m:uri>http://qa-dev.w3.org/wmvs/HEAD/dev/tests/xhtml1-bogus-element.html</m:uri>;
80   *            <m:checkedby>http://validator.w3.org/</m:checkedby>;
81   *            <m:doctype>-//W3C//DTD XHTML 1.0 Transitional//EN</m:doctype>
82   *            <m:charset>utf-8</m:charset>
83   *            <m:validity>false</m:validity>
84   *            <m:errors>
85   *              <m:errorcount>1</m:errorcount>
86   *              <m:errorlist>
87   *              		          
88   *                <m:error>
89   *                  <m:line>13</m:line>
90   *                  <m:col>6</m:col>                                           
91   *                  <m:source><![CDATA[
92   *                    &#60;foo<strong title="Position where error was detected.">&#62;</strong>This phrase is enclosed in a bogus FOO element.&#60;/foo&#62;
93   *                  ]]>
94   *                  </m:source>                                           
95   *                  <m:explanation>
96   *                    <![CDATA[
97   *                    <p> ... </p<p>
98   *                    ]]>
99   *                  </m:explanation>                                           
100  *                  <m:messageid>76</m:messageid>                                           
101  *                  <m:message>element "foo" undefined</m:message>
102  *               </m:error>
103  *          </m:errorlist>
104  *        </m:errors>
105  *        <m:warnings>
106  *          <m:warningcount>0</m:warningcount>
107  *          <m:warninglist>	        
108  *            ...		        
109  *          </m:warninglist>
110  *        </m:warnings>
111  *      </m:markupvalidationresponse>
112  *     </env:Body>
113  *  </env:Envelope>
114  *		}
115  * </pre>
116  *		the structure of this W3CValidator class is aligned to this format
117  */
118 @XmlRootElement(name = "Envelope", namespace = "http://www.w3.org/2003/05/soap-envelope")
119 // no getters/setters are used - all fields are initialized to make this safe
120 // setting should only be done via the check function

140 	/**
141 	 * create a W3CValidator result for the default url http://validator.w3.org/check  with the given html
142 	 * 
143 	 * @param html - the html code to be checked
144 	 * @return - a W3CValidator response according to the SOAP response format or null if the
145 	 * http response status of the Validation service is other than 200
146 	 * explained at response http://validator.w3.org/docs/api.html#requestformat 
147 	 * @throws JAXBException if there is something wrong with the response message so that it
148 	 * can not be unmarshalled
149 	 */
150 	public static W3CValidator check(String html) throws JAXBException {
151 		W3CValidator result=check(url,html);
152 		return result;
153 	}

155 	/**
156 	 * create a W3CValidator result for the given url with the given html
157 	 * 
158 	 * @param url - the url of the validator e.g. "http://validator.w3.org/check"
159 	 * @param html - the html code to be checked
160 	 * @return - a W3CValidator response according to the SOAP response format or null if the
161 	 * http response status of the Validation service is other than 200
162 	 * explained at response http://validator.w3.org/docs/api.html#requestformat 
163 	 * @throws JAXBException if there is something wrong with the response message so that it
164 	 * can not be unmarshalled
165 	 */
166 	public static W3CValidator check(String url, String html)
167 			throws JAXBException {
168 		// initialize the return value
169 		W3CValidator result = null;
170 		
171 		// create a WebResource to access the given url
172 		WebResource resource = Client.create().resource(url);
173 		
174 		// prepare form data for posting
175 		FormDataMultiPart form = new FormDataMultiPart();
176 		
177 		// set the output format to soap12
178 		// triggers the various outputs formats of the validator. If unset, the usual Web format will be sent. 
179 		// If set to soap12, 
180 		// the SOAP1.2 interface will be triggered. See the SOAP 1.2 response format description at
181 		//  http://validator.w3.org/docs/api.html#requestformat
182 		form.field("output", "soap12");
183 	
184 		// make sure Unicode 0x0 chars are removed from html (if any)
185 		// see https://github.com/WolfgangFahl/w3cValidator/issues/1
186 		Pattern pattern = Pattern.compile("[\\000]*");
187 		Matcher matcher = pattern.matcher(html);
188 		if (matcher.find()) {
189 		   html = matcher.replaceAll("");
190 		}
191 		
192 		// The document to validate, POSTed as multipart/form-data
193 		FormDataBodyPart fdp = new FormDataBodyPart("uploaded_file",
194 				IOUtils.toInputStream(html),
195 				// new FileInputStream(tmpHtml),
196 				MediaType.APPLICATION_OCTET_STREAM_TYPE);
197 		
198 		// attach the inputstream as upload info to the form
199 		form.bodyPart(fdp);
200 		
201 		// now post the form via the Internet/Intranet
202 		ClientResponse response = resource.type(MediaType.MULTIPART_FORM_DATA)
203 				.post(ClientResponse.class, form);
204 		// in debug mode show the response status
205 		if (debug)
206 			LOGGER.log(Level.INFO, "response status for '"+url+"'=" + response.getStatus());
207 		// if the http Status is ok
208 		if (response.getStatus() == 200) {
209 			// get the XML encoded SOAP 1.2 response format
210 			String responseXml = response.getEntity(String.class);
211 			// in debug mode show the full xml 
212 			if (debug)
213 				LOGGER.log(Level.INFO,responseXml);
214 			// unmarshal the xml message to the format to a W3CValidator Java object
215 			JAXBContext context = JAXBContext.newInstance(W3CValidator.class);
216 			Unmarshaller u = context.createUnmarshaller();
217 			StringReader xmlReader = new StringReader(responseXml);
218 			// this step will convert from xml text to Java Object
219 			result = (W3CValidator) u.unmarshal(xmlReader);
220 		}
221 		// return the result which might be null if the response status was other than 200
222 		return result;
223 
224 	} // check

242 	 * structure for the Body node of the message

247 	 * </env:Body>
248 	 * }
249 	 * </pre>

256 		@XmlElement(name = "markupvalidationresponse", namespace = "http://www.w3.org/2005/10/markup-validator")

259 		/**
260 		 * The main element of the validation response. Encloses all other information about the validation results.
261 		 * @author wf
262 		 * <pre>
263 		 * {@code
264 		 * <m:markupvalidationresponse env:encodingStyle="http://www.w3.org/2003/05/soap-encoding" xmlns:m="http://www.w3.org/2005/10/markup-validator">
265 		 * ...
266 		 * </m:markupvalidationresponse>
267 		 * }
268 		 * </pre>
269 		 */
270 		@XmlAccessorType(XmlAccessType.FIELD)
271 		public static class ValidationResponse {
272 			/**
273 			 * the address of the document validated. Will (likely?) be upload://Form Submission if an uploaded document or fragment was validated. In EARL terms, this is the TestSubject. 
274 			 */
275 			@XmlElement(namespace = "http://www.w3.org/2005/10/markup-validator")
276 			public String uri;
277 
278 			/**
279 			 * Location of the service which provided the validation result. In EARL terms, this is the Assertor
280 			 */
281 			@XmlElement(namespace = "http://www.w3.org/2005/10/markup-validator")
282 			public String checkedby;
283 			
284 			/**
285 			 * Detected (or forced) Document Type for the validated document
286 			 */
287 			@XmlElement(namespace = "http://www.w3.org/2005/10/markup-validator")
288 			public String doctype;
289 
290 			/**
291 			 * Detected (or forced) Character Encoding for the validated document
292 			 */
293 			@XmlElement(namespace = "http://www.w3.org/2005/10/markup-validator")
294 			public String charset;
295 
296 			/**
297 			 * Whether or not the document validated passed or not formal validation (true|false boolean)
298 			 */
299 			@XmlElement(namespace = "http://www.w3.org/2005/10/markup-validator")
300 			public boolean validity;
301 
302 			/**
303 			 * Encapsulates all data about errors encountered through the validation process
304 			 * <pre>
305        * {@code
306 			 * <m:errors>
307 			 * ...
308 			 * </m:errors>
309 			 * }
310 			 * </pre>
311 			 */
312 			@XmlElement(namespace = "http://www.w3.org/2005/10/markup-validator")
313 			public Errors errors = new Errors();
314 
315 			/**
316 			 * wrapped list of validation errors structurally equivalent to Warnings
317 			 */
318 			public static class Errors {
319 				/**
320 				 * a child of errors, counts the number of errors listed
321 				 */
322 				@XmlElement(namespace = "http://www.w3.org/2005/10/markup-validator")
323 				public int errorcount;
324 				
325 				/**
326 				 * a child of errors, contains the list of errors (surprise!)
327 				 */
328 				@XmlElementWrapper(name = "errorlist", namespace = "http://www.w3.org/2005/10/markup-validator")
329 				@XmlElement(name = "error", namespace = "http://www.w3.org/2005/10/markup-validator")
330 				public List<ValidationError> errorlist = new ArrayList<ValidationError>();
331 
332 			 	/**
333 			 	 * a child of errorlist, contains the information on a single validation error. 
334 			 	 */
335 				@XmlRootElement(name = "error", namespace = "http://www.w3.org/2005/10/markup-validator")
336 				public static class ValidationError extends ValidationAtom {
337 				} // Error
338 			} // Errors
339 
340 			/**
341 			 * Encapsulates all data about warnings encountered through the validation process
342 			 * <pre>
343        * {@code
344 			 * <m:warnings>
345 			 * ...
346 			 * </m:warnings>
347 			 * }
348 			 * </pre>
349 			 */
350 			@XmlElement(namespace = "http://www.w3.org/2005/10/markup-validator")
351 			public Warnings warnings = new Warnings();
352 
353 			/**
354 			 * wrapped list of validation warnings structurally equivalent to Errors
355 			 */
356 			public static class Warnings {
357 				/**
358 				 * a child of warnings, counts the number of warnings listed
359 				 */
360 				@XmlElement(namespace = "http://www.w3.org/2005/10/markup-validator")
361 				public int warningcount;
362 				
363 				/**
364 				 * a child of warnings, contains the list of warnings (surprise!)
365 				 */
366 				@XmlElementWrapper(name = "warninglist", namespace = "http://www.w3.org/2005/10/markup-validator")
367 				@XmlElement(name = "warning", namespace = "http://www.w3.org/2005/10/markup-validator")
368 				public List<ValidationWarning> warninglist = new ArrayList<ValidationWarning>();
369 
370 				/**
371 			 	 * a child of warninglist, contains the information on a single validation warning. 
372 			 	 */
373 				@XmlRootElement(name = "warning", namespace = "http://www.w3.org/2005/10/markup-validator")
374 				public static class ValidationWarning extends ValidationAtom {
375 				} // ValidationWarning
376 			} // Warnings
377 
378 			/**
379 			 * base class for ValidationError and ValidationWarning
380 			 * containing e.g. line, col and message 
381 			 *
382 			 */
383 			public static class ValidationAtom {
384 				/**
385 				 * Within the source code of the validated document, refers to the line where 
386 				 * the error was detected.
387 				 */
388 				@XmlElement(namespace = "http://www.w3.org/2005/10/markup-validator")
389 				public int line;
390 				
391 				/**
392 				 * Within the source code of the validated document, refers to the column of the 
393 				 * line where the error was detected.
394 				 */
395 				@XmlElement(namespace = "http://www.w3.org/2005/10/markup-validator")
396 				public int col;
397 				
398 				/**
399 				 * Snippet of the source where the error was found. Given as HTML fragment within CDATA block.
400 				 */
401 				@XmlCDATA
402 				@XmlElement(namespace = "http://www.w3.org/2005/10/markup-validator")
403 				public String source;
404 				
405 				/**
406 				 * Explanation for the error. Given as HTML fragment within CDATA block.
407 				 */
408 				@XmlCDATA
409 				@XmlElement(namespace = "http://www.w3.org/2005/10/markup-validator")
410 				public String explanation;
411 				
412 				/**
413 				 * The number/identifier of the error, as addressed internally by the validator
414 				 */
415 				@XmlElement(namespace = "http://www.w3.org/2005/10/markup-validator")
416 				public int messageid;
417 				
418 				/**
419 				 * The actual error message
420 				 */
421 				@XmlElement(namespace = "http://www.w3.org/2005/10/markup-validator")
422 				public String message;
423 				
424 				/**
425 				 * convert this W3CValidator to a human readable string
426 				 */
427 				public String toString() {
428 					String kind=this.getClass().getSimpleName();
429 					String result=kind + " line " + line + " col " + col + ":'" + message	+ "'";
430 					return result;
431 				}
432 			} // ValidationAtom
433 		} // ValidationResponse
